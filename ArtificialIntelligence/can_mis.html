\newcommand*\drawitem[6]%
{%
	\ifnum#1=0
	\edef\mylist{}
	\else
	\ifnum#1=1
	\edef\mylist{0,1}
	\else
	\edef\mylist{1,...,#1}
	\fi 
	\foreach \x in \mylist
	{\draw[color=#4, xscale=0.5,yscale=0.5] (#2)
		node[draw=none, xshift=-\x*0.20cm + #6, yshift=#5, font=\small] (s\x) {\tiny{#3}};}
	\fi
}
